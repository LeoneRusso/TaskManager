function applyLang(){
  const L=LANGS[UI.lang]||LANGS.es;
  // All data-i18n elements
  document.querySelectorAll('[data-i18n]').forEach(el=>{
    const key=el.getAttribute('data-i18n');
    if(L[key]!==undefined)el.textContent=L[key];
  });
  // Placeholders
  document.querySelectorAll('[data-i18n-placeholder]').forEach(el=>{
    const key=el.getAttribute('data-i18n-placeholder');
    if(L[key]!==undefined)el.placeholder=L[key];
  });
  // New task button
  document.getElementById('btnNewTask').textContent=L.new_task;
  // Rebuild days picker with localized names
  buildDaysPicker();
  // Update pomodoro label
  if(PS.mode==='work')document.getElementById('pomoLbl').textContent=L.focus;
  else document.getElementById('pomoLbl').textContent=L.rest;
  // Re-render to update dynamic content
  render();
}

function setLang(lang,btn){
  UI.lang=lang;
  localStorage.setItem('tm_lang',lang);
  document.querySelectorAll('.lang-btn').forEach(b=>b.classList.remove('active'));
  btn.classList.add('active');
  applyLang();
}

function loadLang(){
  const saved=localStorage.getItem('tm_lang')||'es';
  UI.lang=saved;
  // Mark active button
  document.querySelectorAll('.lang-btn').forEach(b=>{
    if(b.getAttribute('onclick').includes(`'${saved}'`))b.classList.add('active');
    else b.classList.remove('active');
  });
}

function buildDaysPicker(){
  const L=LANGS[UI.lang]||LANGS.es;
  const el=document.getElementById('daysPickBtns');
  if(!el)return;
  el.innerHTML=L.dn.map((d,i)=>`<button type="button" class="day-btn ${UI.selDays.includes(i)?'sel':''}" data-day="${i}" onclick="toggleDay(this)">${d}</button>`).join('');
}

  SB[key]=!SB[key];
  const body=document.getElementById('sb-'+key);
  const chev=document.getElementById('chev-'+key);
  if(body){body.classList.toggle('closed',!SB[key]);}

  });
}

// ── NEUTRAL XP CALCULATION ──
// Tolerance zone: within tolMin–tolMax (or target) = full baseXP
// Beyond tolMax (good direction) = exponential BONUS
// Below tolMin (bad direction) = exponential PENALTY
function calcNeuXP(task,realVal){
  const nc=task.neutral;
  if(!nc)return 0;
  const sm=getStrMult();
  // Get baseXP from task config (user-defined or default)
  const baseXP=nc.baseXP||Math.round(task.importance*1.5);

  if(nc.type==='value'){
    const target=parseFloat(nc.target)||1;
    const real=parseFloat(realVal)||0;
    const moreGood=nc.dir==='more';

    if(nc.useTol&&nc.tolMin!==undefined&&nc.tolMax!==undefined){
      const lo=parseFloat(nc.tolMin);
      const hi=parseFloat(nc.tolMax);
      if(real>=lo&&real<=hi){
        // Within zone → full base XP
        return Math.round(baseXP*sm);
      }else if(real>hi){
        // Above max → bonus (exponential), regardless of direction preference
        const excess=(real-hi)/Math.max(1,hi-lo);
        const bonus=Math.round(baseXP*(1+Math.pow(excess,0.65)*1.5)*sm);
        return Math.min(baseXP*3,bonus);
      }else{
        // Below min → penalty (exponential)
        const deficit=(lo-real)/Math.max(1,hi-lo);
        const penalty=Math.round(baseXP*(1-Math.pow(deficit,0.65)*1.2)*sm);
        return Math.max(-baseXP*2,penalty);
      }
    }else{
      // No tolerance zone: distance from target = XP modifier
      const diff=real-target;
      const pctDiff=diff/Math.max(1,Math.abs(target));
      const signed=moreGood?pctDiff:-pctDiff;
      if(Math.abs(signed)<0.01)return Math.round(baseXP*sm); // exact
      if(signed>0){
        const bonus=Math.round(baseXP*(1+Math.pow(signed,0.65)*1.5)*sm);
        return Math.min(baseXP*3,bonus);
      }else{
        const penalty=Math.round(baseXP*(1+Math.pow(signed,1.2))*sm);
        return Math.max(-baseXP*2,penalty);
      }
    }
  }

  if(nc.type==='time'){
    const [th,tm]=nc.target.split(':').map(Number);
    const targetMin=th*60+tm;
    const parts=String(realVal).split(':').map(Number);
    if(parts.length<2||isNaN(parts[0]))return 0;
    const realMin=parts[0]*60+parts[1];
    const diffMin=realMin-targetMin; // positive=late, negative=early
    const tol=parseInt(nc.tolerance)||15;
    const absDiff=Math.abs(diffMin);
    if(absDiff<=tol)return Math.round(baseXP*sm); // within tolerance = full XP
    const earlyGood=nc.tdir==='early';
    const good=earlyGood?(diffMin<0):(diffMin>0);
    const overDiff=absDiff-tol;
    if(good){
      const bonus=Math.round(baseXP*(1+Math.pow(overDiff/tol,0.65)*1.2)*sm);
      return Math.min(baseXP*3,bonus);
    }else{
      const penalty=Math.round(baseXP*(1-Math.min(1.8,Math.pow(overDiff/tol,0.8)))*sm);
      return Math.max(-baseXP,Math.round(penalty));
    }
  }

  if(nc.type==='date'){
    const target=new Date(nc.dateTarget+'T12:00:00');
    const real=new Date(realVal+'T12:00:00');
    if(isNaN(real))return 0;
    const diffDays=Math.round((real-target)/(864e5)); // positive=late, negative=early
    if(diffDays===0)return Math.round(baseXP*sm); // exact = full baseXP
    const earlyGood=nc.ddir==='early';
    const good=earlyGood?(diffDays<0):(diffDays>0);
    const absDays=Math.abs(diffDays);
    if(good){
      const bonus=Math.round(baseXP*(1+Math.pow(absDays*0.25,0.7))*sm);
      return Math.min(baseXP*3,bonus);
    }else{
      const penalty=Math.round(baseXP*(1-Math.min(1.8,Math.pow(absDays*0.35,0.8)))*sm);
      return Math.max(-baseXP*2,Math.round(penalty));
    }
  }
  return 0;
}

function completeTask(id){
  const tk=S.tasks.find(x=>x.id===id);if(!tk||tk.done)return;
  const xpAmt=Math.round(tk.importance*1.5);const sm=getStrMult();
  const finalXP=tk.negative?-xpAmt:Math.round(xpAmt*sm);
  const now=new Date();
  tk.done=true;tk.lastDone=now.toISOString();
  if(tk.negative){S.negXP+=xpAmt;S.negDone++;}else{S.done++;}
  S.xp=Math.max(0,S.xp+finalXP);
  if(!tk.negative){
    const today=todayS();
    if(S.lastDoneDate!==today){
      const y=new Date();y.setDate(y.getDate()-1);const yS=y.toISOString().slice(0,10);
      S.streak=(S.lastDoneDate===yS)?S.streak+1:1;S.lastDoneDate=today;
    }
  }
  // Track special achievement stats
  const todayKey=todayS();
  const hr=now.getHours();
  if(hr>=0&&hr<3)S.nightOwl=true;
  if(hr>=4&&hr<6)S.earlyBird=true;
  if(!S.tasksToday||S.tasksToday.date!==todayKey)S.tasksToday={date:todayKey,count:0};
  S.tasksToday.count++;
  if(S.tasksToday.count>(S.maxTasksInDay||0))S.maxTasksInDay=S.tasksToday.count;
  if(!S.xpToday||S.xpToday.date!==todayKey)S.xpToday={date:todayKey,total:0};
  if(finalXP>0){S.xpToday.total+=finalXP;if(S.xpToday.total>(S.maxXPInDay||0))S.maxXPInDay=S.xpToday.total;}
  if(tk.negative){S.lastNegDate=todayKey;}
  S.history.unshift({id:Date.now(),taskName:tk.name,icon:tk.icon,xp:finalXP,negative:tk.negative,date:now.toISOString(),sm,taskId:tk.id});
  if(S.history.length>200)S.history=S.history.slice(0,200);
  saveS();updateAll();render();
  if(tk.negative)toast('xpl','\u26a0\ufe0f',`\u2212${xpAmt} XP \u00b7 ${tk.name}`);
  else toast('xpg','\u26a1',`+${finalXP} XP${sm>1?` (\u00d7${sm}!)`:''} \u00b7 ${tk.name}`);
  checkAch();
}
function getStrMult(){if(S.streak>=30)return 3;if(S.streak>=14)return 2.5;if(S.streak>=7)return 2;if(S.streak>=3)return 1.5;return 1;}
function deleteTask(id){
  const tk=S.tasks.find(x=>x.id===id);
  if(!tk)return;
  const msg=(t('confirm_delete')||'Delete?')+'\n'+tk.name;
  if(!window.confirm(msg))return;
  // If task was done, revert its XP contribution from history
  if(tk.done){
    const histIdx=S.history.findIndex(h=>h.taskId===tk.id&&!h.penalty);
    if(histIdx>=0){
      const h=S.history[histIdx];
      // Revert XP
      S.xp=Math.max(0,S.xp-h.xp);
      if(h.xp>0)S.done=Math.max(0,S.done-1);
      else if(h.xp<0)S.negDone=Math.max(0,S.negDone-1);
      // Check if this was the ONLY task done today → revert streak
      const taskDate=h.date.slice(0,10);
      const todayTasks=S.history.filter((x,i)=>i!==histIdx&&x.date.slice(0,10)===taskDate&&!x.penalty&&x.xp>0);
      if(todayTasks.length===0&&S.lastDoneDate===taskDate){
        S.streak=Math.max(0,S.streak-1);
        // Recalculate lastDoneDate to the previous day that had activity
        const prevHist=S.history.filter((x,i)=>i!==histIdx&&!x.penalty&&x.xp>0&&x.date.slice(0,10)<taskDate);
        S.lastDoneDate=prevHist.length?prevHist[0].date.slice(0,10):null;
      }
      S.history.splice(histIdx,1);
    }
  }
  S.tasks=S.tasks.filter(x=>x.id!==id);
  saveS();render();updateAll();
}

function getFiltered(){
  const q=document.getElementById('searchInput').value.toLowerCase();
  const sort=document.getElementById('sortSel').value;
  const today=new Date();
  let tasks=[...S.tasks].filter(tk=>{
    const isNeu=!!tk.neutral&&!tk.negative;
    if(q&&!tk.name.toLowerCase().includes(q)&&!(tk.notes||'').toLowerCase().includes(q))return false;
    if(UI.gf!=='all'&&tk.group!==UI.gf)return false;
    if(UI.polarity==='pos'&&(tk.negative||isNeu))return false;
    if(UI.polarity==='neg'&&!tk.negative)return false;
    if(UI.polarity==='neu'&&!isNeu)return false;
    if(UI.dayFilter==='today'&&!isTaskToday(tk))return false;
    if(UI.filter==='done')return tk.done;
    if(UI.filter==='negative')return tk.negative;
    if(UI.filter==='neutral')return isNeu;
    if(UI.filter==='progress')return!!tk.progress;
    if(UI.filter!=='all')return tk.repeat===UI.filter;
    return true;
  });
  tasks.sort((a,b)=>{
    if(sort==='importance')return b.importance-a.importance;
    if(sort==='name')return a.name.localeCompare(b.name);
    if(sort==='xp')return(b.importance*1.5)-(a.importance*1.5);
    if(sort==='created')return new Date(b.createdAt)-new Date(a.createdAt);

    return 0;
  });
  return tasks;
}

function undoHistXP(idx){
  const h=S.history[idx];if(!h)return;
  const xp=h.xp;
  const taskDate=h.date.slice(0,10);
  // Revert XP and counters
  S.xp=Math.max(0,S.xp-xp);
  if(xp>0){S.done=Math.max(0,S.done-1);if(h.neutral)S.neuDone=Math.max(0,(S.neuDone||0)-1);}
  else if(xp<0&&!h.penalty)S.negDone=Math.max(0,S.negDone-1);
  // Revert streak if this was the only positive task today
  if(xp>0&&!h.penalty){
    const otherTasksToday=S.history.filter((x,i)=>i!==idx&&x.date.slice(0,10)===taskDate&&!x.penalty&&x.xp>0);
    if(otherTasksToday.length===0&&S.lastDoneDate===taskDate){
      S.streak=Math.max(0,S.streak-1);
      const prevHist=S.history.filter((x,i)=>i!==idx&&!x.penalty&&x.xp>0&&x.date.slice(0,10)<taskDate);
      S.lastDoneDate=prevHist.length?prevHist[0].date.slice(0,10):null;
    }
  }
  // Mark the task as undone if it still exists
  if(h.taskId){
    const tk=S.tasks.find(x=>x.id===h.taskId);
    if(tk){tk.done=false;tk.lastDone=null;}
  }
  S.history.splice(idx,1);
  saveS();updateAll();render();
  if(UI.view==='history')renderHistory();
  toast('inf','↩',t('hist_undone'));
}

// ── HELPER: Is a task active on a given date? ──
function taskActiveOnDate(tk, dateObj) {
  const dateStr = dateObj.toISOString().slice(0, 10);
  const dayOfWeek = dateObj.getDay(); // 0=Sun … 6=Sat
  const month = dateObj.getMonth();
  const day = dateObj.getDate();

  if (tk.repeat === 'daily') return true;
  if (tk.repeat === 'once') {
    // Once tasks with deadline: active from creation date through deadline date
    if (tk.deadline) {
      const createdStr = tk.createdAt ? tk.createdAt.slice(0, 10) : dateStr;
      return dateStr >= createdStr && dateStr <= tk.deadline;
    }
    // Once tasks without deadline: active only on creation date
    if (tk.createdAt && tk.createdAt.slice(0, 10) === dateStr) return true;
    return false;
  }
  if (tk.repeat === 'weekly') {
    return !!(tk.weekDays && tk.weekDays.includes(dayOfWeek));
  }
  if (tk.repeat === 'annual') {
    const created = tk.createdAt ? new Date(tk.createdAt) : null;
    if (created) return created.getMonth() === month && created.getDate() === day;
    return false;
  }
  // Also include tasks with a deadline on this specific date
  if (tk.deadline && tk.deadline === dateStr) return true;
  return false;
}

function isTaskToday(tk) {
  return taskActiveOnDate(tk, new Date());
}

function autoReset(){
  const today=todayS();const todayDay=new Date().getDay();
  S.tasks.forEach(tk=>{if(!tk.done||!tk.lastDone)return;const ld=tk.lastDone.slice(0,10);
    if(tk.repeat==='daily'&&ld!==today){tk.done=false;}

  });
  // Update noNegDays
  if(S.lastNegDate){
    const last=new Date(S.lastNegDate+'T12:00:00');
    const now=new Date(today+'T12:00:00');
    S.noNegDays=Math.max(0,Math.round((now-last)/(864e5)));
  } else {
    // Never had a negative: count from first history entry
    const firstHist=S.history.length?S.history[S.history.length-1]:null;
    if(firstHist){
      const first=new Date(firstHist.date);
      const now=new Date(today+'T12:00:00');
      S.noNegDays=Math.max(0,Math.round((now-first)/(864e5)));
    }
  }
  saveS();
}

// ── DAILY INACTIVITY PENALTY ──
// Each day with no completed task: lose 100 XP × level
function applyDailyPenalty(){
  const today=todayS();
  if(S.lastPenaltyDate===today)return; // already checked today
  if(!S.lastDoneDate)return; // never done anything
  // Find how many full days since last activity
  const last=new Date(S.lastDoneDate+'T12:00:00');
  const now=new Date(today+'T12:00:00');
  const daysMissed=Math.round((now-last)/(864e5));
  if(daysMissed<=0)return; // did something today
  const level=calcLevel(S.xp);
  const penaltyPerDay=level*100;
  const totalPenalty=penaltyPerDay*daysMissed;
  S.xp=Math.max(0,S.xp-totalPenalty);
  S.streak=0;
  S.lastPenaltyDate=today;
  // Add to history
  for(let i=0;i<Math.min(daysMissed,7);i++){
    const d=new Date(last);d.setDate(d.getDate()+i+1);
    S.history.unshift({id:Date.now()+i,taskName:`😴 ${t('penalty_toast')||'Sin actividad'}`,icon:'😴',xp:-penaltyPerDay,negative:true,penalty:true,date:d.toISOString(),sm:1});
  }
  if(S.history.length>200)S.history=S.history.slice(0,200);
  saveS();
  if(totalPenalty>0){
    setTimeout(()=>toast('xpl','😴',`${t('penalty_toast')||'Sin actividad'} −${totalPenalty} XP (${daysMissed}d × Nv${level})`),800);
  }
}

function calcLevel(xp){
  // Level thresholds: 0, 500, 1500, 3000, 5000, 8000, ...
  if(xp<500)return 1;
  if(xp<1500)return 2;
  if(xp<3000)return 3;
  if(xp<5000)return 4;
  if(xp<8000)return 5;
  if(xp<12000)return 6;
  if(xp<17000)return 7;
  if(xp<23000)return 8;
  if(xp<30000)return 9;
  return Math.floor(xp/5000)+1;
}

function loadAvatarFile(event) {
  const file = event.target.files[0];
  if (!file) return;
  const reader = new FileReader();
  reader.onload = function(e) {
    const img = new Image();
    img.onload = function() {

      _cropOffX = 0;
      _cropOffY = 0;
      document.getElementById('avtZoom').value = 1;
      document.getElementById('avtCropWrap').style.display = 'block';
      document.getElementById('avtUploadArea').style.display = 'none';
      drawCropCanvas();
      setupCropDrag();
    };
    img.src = e.target.result;
  };
  reader.readAsDataURL(file);
}

function drawCropCanvas() {
  if (!_cropImg) return;
  const canvas = document.getElementById('avtCropCanvas');
  const ctx = canvas.getContext('2d');
  const size = 200;
  const zoom = parseFloat(document.getElementById('avtZoom').value) || 1;

  // Compute draw dimensions keeping aspect ratio
  const iw = _cropImg.naturalWidth, ih = _cropImg.naturalHeight;
  const scale = Math.max(size / iw, size / ih) * zoom;
  const dw = iw * scale, dh = ih * scale;

  // Clamp offsets so image always covers the canvas
  _cropOffX = Math.min(0, Math.max(size - dw, _cropOffX));
  _cropOffY = Math.min(0, Math.max(size - dh, _cropOffY));

  ctx.clearRect(0, 0, size, size);
  // Clip to circle
  ctx.save();
  ctx.beginPath();
  ctx.arc(size/2, size/2, size/2, 0, Math.PI*2);
  ctx.clip();
  ctx.drawImage(_cropImg, _cropOffX, _cropOffY, dw, dh);
  ctx.restore();
}

function setupCropDrag() {
  const canvas = document.getElementById('avtCropCanvas');
  canvas.onmousedown = canvas.ontouchstart = function(e) {
    _cropDragging = true;
    const pos = e.touches ? e.touches[0] : e;
    _cropStartX = pos.clientX - _cropOffX;
    _cropStartY = pos.clientY - _cropOffY;
    e.preventDefault();
  };
  canvas.onmousemove = canvas.ontouchmove = function(e) {
    if (!_cropDragging) return;
    const pos = e.touches ? e.touches[0] : e;
    _cropOffX = pos.clientX - _cropStartX;
    _cropOffY = pos.clientY - _cropStartY;
    drawCropCanvas();
    e.preventDefault();
  };
  canvas.onmouseup = canvas.ontouchend = () => { _cropDragging = false; };
}
